[tex]\tt{10x\:=\:120}[/tex]

[tex]\large\tt{ \frac{10x}{10} \:=\: \frac{120}{10}}[/tex]

[tex]\tt{x\:=\:12}[/tex]

[tex]\tt{BR\:=\:BC\:+\:CR}[/tex]

[tex]\tt{BR\:=\:12\:+\:8}[/tex]
