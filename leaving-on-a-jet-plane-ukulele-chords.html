Given:s = ?d = 4,500 kilometerst = 5 hours

[tex]\large \sf \blue{speed = \frac{distance}{time} } \\ \large \sf \blue{speed = \frac{4 \: 500 \: km}{5 \: hrs}} \\ \large \sf \blue{speed = 900 \: kph }[/tex]
