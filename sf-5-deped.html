[tex] \underline{ \sf{ \pmb{Given \: integral }\: is : }}[/tex]

[tex]\begin{gathered} \quad \hookrightarrow \: \boxed{\bold{\displaystyle{ \bold{\int\bold{ {x}^{5} {e}^{ - {4x}^{3} } \: dx \: = \: \dfrac{1}{48}{e}^{ - {4x}^{3} } + c }}}}}\\ \end{gathered} [/tex]
